testcases:
  - name: control_flow_demo
    tags: [gui, control-flow]
    on_step_failure: stop
    steps:
      - set:
          var: channels
          value: [1, 2, 3]
      - for:
          each: "${channels}"
          as: channel
          do:
            - log.text:
                text: "Turn on {{ channel }}"
            - if:
                condition: "${channel == 2}"
                then:
                  - for:
                      each: "${retries}"
                      as: retry
                      do:
                        - delay:
                            ms: 250
            - delay:
                ms: 1000
      - assert.eq:
          left: "${voltage_ok}"
          right: true
